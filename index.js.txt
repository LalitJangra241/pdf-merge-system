const express = require("express");
const { PDFDocument } = require("pdf-lib");
const axios = require("axios");

const app = express();
app.use(express.json({ limit: "50mb" }));

// Fetch PDF from URL
async function fetchPDF(url) {
  const res = await axios.get(url, { responseType: "arraybuffer" });
  return res.data;
}

// Merge API
app.post("/merge", async (req, res) => {
  try {
    const { pdfLinks } = req.body;

    if (!pdfLinks || pdfLinks.length === 0) {
      return res.status(400).json({ error: "No PDFs provided" });
    }

    const mergedPdf = await PDFDocument.create();

    for (let link of pdfLinks) {
      const pdfBytes = await fetchPDF(link);
      const pdf = await PDFDocument.load(pdfBytes);

      const pages = await mergedPdf.copyPages(pdf, pdf.getPageIndices());
      pages.forEach(p => mergedPdf.addPage(p));
    }

    const finalPdf = await mergedPdf.save();

    res.json({
      mergedPdf: Buffer.from(finalPdf).toString("base64")
    });

  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Merge failed" });
  }
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("Server running on port " + PORT));
